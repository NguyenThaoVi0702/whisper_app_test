vietnamese_chars = "a-zàáâãèéêìíòóôõùúýăđĩũơưạảấầẩẫậắằẳẵặẹẻẽếềểễệỉịọỏốồổỗộớờởỡợụủứừửữự"
numbers = "0-9"

def normalize_transcription(text):
    """
    Normalizes transcription text by:
    1. Converting to lowercase.
    2. Removing all characters except for Vietnamese letters, numbers, and spaces.
    3. Collapsing multiple spaces into a single space.
    """
    text = text.lower()
    text = re.sub(f'[^{vietnamese_chars}{numbers}\s]', '', text)
    text = re.sub(r'\s+', ' ', text).strip()
    return text
